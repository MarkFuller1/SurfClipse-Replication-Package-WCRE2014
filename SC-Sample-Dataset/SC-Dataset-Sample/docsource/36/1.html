public class SqliteHelper {

private Connection connection = null;
private Statement statement = null;
private String _dbName = "";
private static Log log = null;

private QueryRunner qryRun = null;

public SqliteHelper(String dbname) throws ClassNotFoundException {
    Class.forName("org.sqlite.JDBC");
    _dbName = dbname;
    log = LogHelper.getLog();
    log.info("sqlite dbpath:" + dbname);
    if(connection == null) {
        connectToSqlite();
    }
}

public void connectToSqlite() {
    try {
        connection = DriverManager.getConnection("jdbc:sqlite:" + _dbName);
        statement = connection.createStatement();
        statement.setQueryTimeout(3); // set time out
        qryRun = new QueryRunner();
        log.info("init sqlite connection...");
    } catch (SQLException e) {
        log.error(e.getMessage());
    }
}

// set change new object
public Map<?, ?> select(String query, Object[] where) {
    try {
        log.info("run sql:" + query);
        return (Map<?, ?>) qryRun.query(connection, query,
                new MapHandler(), where);
    } catch (SQLException e) {
        e.printStackTrace();
        log.error(e.getMessage());
    }finally{
        cleanup();
    }
    return null;
}

// return map list
public List<?> selectAll(String query) {
    try {
        return (List<?>) qryRun.query(connection,query, new MapListHandler());
    } catch (SQLException e) {
        e.printStackTrace();
        log.error(e.getMessage());
    }finally{
        cleanup();
    }
    return null;
}

// get first column
public String column(String query, String field, Object[] where) {
    String ret = null;
    try {
        List<?> lField = (List<?>) qryRun.query(connection, query,
                new ColumnListHandler<Map>(field), where);
        for (int i = 0; i < lField.size();) {
            ret = (String) lField.get(i);
            break;
        }
    } catch (SQLException e) {
        e.printStackTrace();
        log.error(e.getMessage());
    }finally{
        cleanup();
    }
    return ret;
}

// get column list
public List<?> columnAll(String query, String field, Object[] where) {
    try {
        return (List<?>) qryRun.query(connection, query,
                new ColumnListHandler<Map>(field), where);
    } catch (SQLException e) {
        e.printStackTrace();
        log.error(e.getMessage());
    }finally{
        cleanup();
    }
    return null;
}

public int exec(String query, Object[] obj) {
    try {
        return qryRun.update(connection, query, obj);
    } catch (SQLException e) {
        e.printStackTrace();
        try {
            DbUtils.rollback(connection);
        } catch (SQLException re) {
            re.printStackTrace();
            log.error(re.getMessage());
        }
    }finally{
        cleanup();
    }
    return -1;
}

// close db connection
public void cleanup() {
    DbUtils.closeQuietly(connection);
}

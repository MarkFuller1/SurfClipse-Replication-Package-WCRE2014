  23.     /**
  24.      * Provides a way to determine the next visually represented model 
  25.      * location that one might place a caret.  Some views may not be visible,
  26.      * they might not be in the same order found in the model, or they just
  27.      * might not allow access to some of the locations in the model.
  28.      * <p>
  29.      * This implementation assumes the views are layed out in a logical
  30.      * manner. That is, that the view at index x + 1 is visually after
  31.      * the View at index x, and that the View at index x - 1 is visually
  32.      * before the View at x. There is support for reversing this behavior
  33.      * only if the passed in <code>View</code> is an instance of
  34.      * <code>CompositeView</code>. The <code>CompositeView</code>
  35.      * must then override the <code>flipEastAndWestAtEnds</code> method.
  36.      *
  37.      * @param v View to query
  38.      * @param pos the position to convert >= 0
  39.      * @param a the allocated region to render into
  40.      * @param direction the direction from the current position that can
  41.      *  be thought of as the arrow keys typically found on a keyboard;
  42.      *  this may be one of the following: 
  43.      *  <ul>
  44.      *  <li><code>SwingConstants.WEST</code>
  45.      *  <li><code>SwingConstants.EAST</code> 
  46.      *  <li><code>SwingConstants.NORTH</code>
  47.      *  <li><code>SwingConstants.SOUTH</code>  
  48.      *  </ul>
  49.      * @param biasRet an array contain the bias that was checked
  50.      * @return the location within the model that best represents the next
  51.      *  location visual position
  52.      * @exception BadLocationException
  53.      * @exception IllegalArgumentException if <code>direction</code> is invalid
  54.      */
  55.     public static int getNextVisualPositionFrom(View v, int pos, Position.Bias b,
  56.                                           Shape alloc, int direction,
  57.                                           Position.Bias[] biasRet)
  58.                              throws BadLocationException {
  59.         if (v.getViewCount() == 0) {
  60.             // Nothing to do.
  61.             return pos;
  62.         }
  63.         boolean top = (direction == SwingConstants.NORTH ||
  64.                        direction == SwingConstants.WEST);
  65.         int retValue;
  66.         if (pos == -1) {
  67.             // Start from the first View.
  68.             int childIndex = (top) ? v.getViewCount() - 1 : 0;
  69.             View child = v.getView(childIndex);
  70.             Shape childBounds = v.getChildAllocation(childIndex, alloc);
  71.             retValue = child.getNextVisualPositionFrom(pos, b, childBounds,
  72.                                                        direction, biasRet);
  73. 	    if (retValue == -1 && !top && v.getViewCount() > 1) {
  74. 		// Special case that should ONLY happen if first view
  75. 		// isn't valid (can happen when end position is put at
  76. 		// beginning of line.
  77. 		child = v.getView(1);
  78.                 childBounds = v.getChildAllocation(1, alloc);
  79. 		retValue = child.getNextVisualPositionFrom(-1, biasRet[0],
  80.                                                            childBounds,
  81.                                                            direction, biasRet);
  82. 	    }
  83.         }
  84.         else {
  85.             int increment = (top) ? -1 : 1;
  86.             int childIndex;
  87.             if (b == Position.Bias.Backward && pos > 0) {
  88.                 childIndex = v.getViewIndex(pos - 1, Position.Bias.Forward);
  89.             }
  90.             else {
  91.                 childIndex = v.getViewIndex(pos, Position.Bias.Forward);
  92.             }
  93.             View child = v.getView(childIndex);
  94.             Shape childBounds = v.getChildAllocation(childIndex, alloc);
  95.             retValue = child.getNextVisualPositionFrom(pos, b, childBounds,
  96.                                                        direction, biasRet);
  97.  
  98.  
  99.  
  100.             if ((direction == SwingConstants.EAST ||
  101.                  direction == SwingConstants.WEST) &&
  102.                 (v instanceof CompositeView) &&
  103.                 flipEastAndWestAtEnds(pos, b)) {
  104.                 increment *= -1;
  105.             }
  106.             childIndex += increment;
  107.             if (retValue == -1 && childIndex >= 0 &&
  108.                                   childIndex < v.getViewCount()) {
  109.                 child = v.getView(childIndex);
  110.                 childBounds = v.getChildAllocation(childIndex, alloc);
  111.                 retValue = child.getNextVisualPositionFrom(
  112.                                      -1, b, childBounds, direction, biasRet);
  113.                 // If there is a bias change, it is a fake position
  114.                 // and we should skip it. This is usually the result
  115.                 // of two elements side be side flowing the same way.
  116.                 if (retValue == pos && biasRet[0] != b) {
  117.                     return getNextVisualPositionFrom(v, pos, biasRet[0],
  118.                                                      alloc, direction,
  119.                                                      biasRet);
  120.                 }
  121.             }
  122.             else if (retValue != -1 && biasRet[0] != b &&
  123.                      ((increment == 1 && child.getEndOffset() == retValue) ||
  124.                       (increment == -1 &&
  125.                        child.getStartOffset() == retValue)) &&
  126.                      childIndex >= 0 && childIndex < v.getViewCount()) {
  127.                 // Reached the end of a view, make sure the next view
  128.                 // is a different direction.
  129.                 child = v.getView(childIndex);
  130.                 childBounds = v.getChildAllocation(childIndex, alloc);
  131.                 Position.Bias originalBias = biasRet[0];
  132.                 int nextPos = child.getNextVisualPositionFrom(
  133.                                     -1, b, childBounds, direction, biasRet);
  134.                 if (biasRet[0] == b) {
  135.                     retValue = nextPos;
  136.                 }
  137.                 else {
  138.                     biasRet[0] = originalBias;
  139.                 }
  140.             }
  141.         }
  142.  
  143.         System.out.println(" retval = " + retValue);
  144.         return retValue;

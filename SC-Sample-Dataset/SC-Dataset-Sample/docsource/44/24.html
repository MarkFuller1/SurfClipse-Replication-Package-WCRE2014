 594:
 595:    // This can come on any thread. If we are in the process of reloading
 596:    // the image and determining our state (loading == true) we don't fire
 597:    // preference changed, or repaint, we just reset the fWidth/fHeight as
 598:    // necessary and return. This is ok as we know when loading finishes
 599:    // it will pick up the new height/width, if necessary.
 600:
 601:    private static boolean sIsInc = true;
 602:    private static int sIncRate = 100;
 603:
 604:    /* (non-Javadoc)
 605:     * @see java.awt.image.ImageObserver#imageUpdate(java.awt.Image, int, int, int, int, int)
 606:     */
 607:    public boolean imageUpdate(Image img, int flags, int x, int y, int width, int height)
 608:...	      {
 609:        if((fImage == null) || (fImage != img))
 610:...	          {
 611:            return false;
 612:        }
 613:
 614:        // Bail out if there was an error
 615:        if((flags & (ABORT|ERROR)) != 0)
 616:...	          {
 617:            fImage = null;
 618:            repaint(0);
 619:            return false;
 620:        }
 621:
 622:        // Resize image if necessary
 623:        short changed = 0;
 624:        if((flags & ImageObserver.HEIGHT) != 0)
 625:...	          {
 626:            if(!getElement().getAttributes().isDefined(HTML.Attribute.HEIGHT))
 627:...	              {
 628:                changed |= 1;
 629:            }
 630:        }
 631:        if((flags & ImageObserver.WIDTH) != 0)
 632:...	          {
 633:            if(!getElement().getAttributes().isDefined(HTML.Attribute.WIDTH))
 634:...	              {
 635:                changed |= 2;
 636:            }
 637:        }
 638:
 639:        synchronized(this)
 640:...	          {
 641:            if((changed & 1) == 1)
 642:...	              {
 643:                fWidth = width;
 644:            }
 645:            if((changed & 2) == 2)
 646:...	              {
 647:                fHeight = height;
 648:            }
 649:            if(bLoading)
 650:...	              {
 651:                // No need to resize or repaint, still in the process of loading
 652:                return true;
 653:            }
 654:        }
 655:
 656:        if(changed != 0)
 657:...	          {
 658:            // May need to resize myself, asynchronously
 659:            Document doc = getDocument();
 660:            try
 661:...	              {
 662:                if(doc instanceof AbstractDocument)
 663:...	                  {
 664:                    ((AbstractDocument)doc).readLock();
 665:                }
 666:                preferenceChanged(this, true, true);
 667:            }
 668:            finally
 669:...	              {
 670:                if(doc instanceof AbstractDocument)
 671:...	                  {
 672:                    ((AbstractDocument)doc).readUnlock();
 673:                }
 674:            }
 675:            return true;
 676:        }
 677:
 678:        // Repaint when done or when new pixels arrive
 679:        if((flags & (FRAMEBITS|ALLBITS)) != 0)
 680:...	          {
 681:            repaint(0);
 682:        }
 683:        else if((flags & SOMEBITS) != 0)
 684:...	          {
 685:            if(sIsInc)
 686:...	              {
 687:                repaint(sIncRate);
 688:            }
 689:        }
 690:        return ((flags & ALLBITS) == 0);
 691:    }

 918:    throws IOException
 919:...	      {
 920:        /* Copy resource into a byte array. This is
 921:         * necessary because several browsers consider
 922:         * Class.getResource a security risk because it
 923:         * can be used to load additional classes.
 924:         * Class.getResourceAsStream just returns raw
 925:         * bytes, which we can convert to an image.
 926:         */
 927:        InputStream resource = RelativeImageView.class.getResourceAsStream(gifFile);
 928:
 929:        if(resource == null)
 930:...	          {
 931:            return null;
 932:        }
 933:        BufferedInputStream in = new BufferedInputStream(resource);
 934:        ByteArrayOutputStream out = new ByteArrayOutputStream(1024);
 935:        byte[] buffer = new byte[1024];
 936:        int n;
 937:        while((n = in.read(buffer)) > 0)
 938:...	          {
 939:            out.write(buffer, 0, n);
 940:        }
 941:        in.close();

 943:

var win = Ti.UI.createWindow({
	backgroundColor : '#00f'
});

var maproute = {
	name : 'runtrackerroute',
	points : [{
		latitude : 42.3353,
		longitude : -71.1715
	}, {
		latitude : 42.3355,
		longitude : -71.1725
	}, {
		latitude : 42.3359,
		longitude : -71.1735
	}],
	color : 'blue',
	width : 4
};

var map = Titanium.Map.createView({
	mapType : Titanium.Map.STANDARD_TYPE,
	region : {
		latitude : 42.3366,
		longitude : -71.1689,
		latitudeDelta : 0.01,
		longitudeDelta : 0.01
	}
});
map.addRoute(maproute);

function UpdateMapRoute(lat, lon)
{
	//remove route
	map.removeRoute(maproute);

	//add new points
	maproute.points.push({
		'latitude' : lat,
		'longitude' : lon
	});

	//add route
	map.addRoute(maproute);
	
	map.setLocation({
		latitude : lat,
		longitude : lon,
		animate : true,
		latitudeDelta : 0.001,
		longitudeDelta : 0.004
	});

}

var lat = 42.3359;
var lon = -71.1735;

setInterval(function(){
	UpdateMapRoute(lat,lon);
	lat = lat + .002;
	lon = lon + .002;
},500);



win.add(map);
win.open();

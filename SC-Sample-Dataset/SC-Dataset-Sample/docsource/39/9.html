Listing All Available Unicode to Character Set Converters
Map map = Charset.availableCharsets();
Iterator it = map.keySet().iterator();
while (it.hasNext()) {
    // Get charset name
    String charsetName = (String)it.next();
    // Get charset
    Charset charset = Charset.forName(charsetName);
}

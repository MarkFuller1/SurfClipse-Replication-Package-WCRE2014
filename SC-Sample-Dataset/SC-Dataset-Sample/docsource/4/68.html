package com.test.server;

import java.io.Serializable;
import java.util.LinkedList;

public abstract class AMessageStrategy implements Serializable {
	private static final long			serialVersionUID	= 6461613636863297714L;
	protected LinkedList<Serializable>		messageList;
	private static int					REQUEST_ID			= 0;
	private static int					DATA				= 1;

	public AMessageStrategy(Integer newRequestId, Serializable newData) {
		messageList = new LinkedList<Serializable>();
		messageList.add(REQUEST_ID, newRequestId); // Request/Response
		messageList.add(DATA, newData);
	}
}
